import makeWASocket, { 
    useMultiFileAuthState, 
    DisconnectReason, 
    fetchLatestWaWebVersion,
    generateWAMessageFromContent,
    prepareWAMessageMedia 
} from '@whiskeysockets/baileys';
import pino from 'pino';
import { Boom } from '@hapi/boom';
import fs from 'fs';
import axios from 'axios';
import dotenv from 'dotenv';
import qrcode from 'qrcode-terminal';

// Load environment variables from the .env file (API keys, phone numbers, etc.)
dotenv.config();

// ==========================================
// ⚙️ CONFIGURATION & GLOBAL STATE
// ==========================================
// Directory where Baileys will store the WhatsApp session data (so you don't scan QR every time)
const AUTH_FOLDER = './session'; 
const OPENROUTER_API_KEY = process.env.OPENROUTER_API_KEY; 
const PAYSTACK_SECRET_KEY = process.env.PAYSTACK_SECRET_KEY;
const SHOP_PHONE = process.env.SHOP_PHONE_NUMBER || "233000000000"; 

// Global instances
let sock = null; // Holds the active WhatsApp socket connection
let isConnecting = false; // Prevents multiple simultaneous connection attempts

// In-memory state management. 
// Keys are WhatsApp numbers (JIDs). Values hold the user's current 'state' (chat, checkout, etc.) and 'cart' data.
// For a larger production system, consider moving this to Redis or a database.
const userSessions = {}; 

// ==========================================
// 🚀 BOT INITIALIZATION & CONNECTION
// ==========================================
async function startBot() {
    if (isConnecting) return;
    isConnecting = true;

    // Ensure the session folder exists to prevent crash on first run
    if (!fs.existsSync(AUTH_FOLDER)) fs.mkdirSync(AUTH_FOLDER, { recursive: true });
    
    // MultiFileAuthState creates a folder of JSON files to manage session keys securely
    const { state, saveCreds } = await useMultiFileAuthState(AUTH_FOLDER);
    const { version } = await fetchLatestWaWebVersion();

    // Initialize the main WhatsApp socket
    sock = makeWASocket({
        version,
        logger: pino({ level: 'silent' }), // Hides the massive Baileys debug logs in terminal
        auth: state,
        browser: ['EllTek Store', 'Chrome', '1.0.0'], // Sets the device name shown in linked devices
        // Fallback for older messages to prevent decryption errors
        getMessage: async (key) => ({ conversation: 'EllTek Support' }) 
    });

    // Connection lifecycle listener (QR Code, Online, Disconnected)
    sock.ev.on('connection.update', async (update) => {
        const { connection, lastDisconnect, qr } = update;
        
        // Print QR code to terminal for the initial login
        if (qr) qrcode.generate(qr, { small: true });
        
        if (connection === 'open') {
            isConnecting = false;
            console.log('✅ EllTek Production Ready - Global Hybrid AI Active!');
        }
        
        // Auto-reconnect logic if the connection drops
        if (connection === 'close') {
            isConnecting = false;
            const statusCode = lastDisconnect?.error instanceof Boom ? lastDisconnect.error.output.statusCode : 0;
            // Reconnect unless the user actively logged out via their phone settings
            if (statusCode !== DisconnectReason.loggedOut) setTimeout(() => startBot(), 5000);
        }
    });

    // Save credentials whenever they are updated to keep the session alive
    sock.ev.on('creds.update', saveCreds);

    // ==========================================
    // 📨 INCOMING MESSAGE HANDLER
    // ==========================================
    sock.ev.on('messages.upsert', async ({ messages, type }) => {
        // Only process new messages, ignore history syncs
        if (type !== 'notify') return;
        const msg = messages[0];
        
        // Ignore empty messages or messages sent by the bot itself
        if (!msg.message || msg.key.fromMe) return;

        const from = msg.key.remoteJid;
        
        // Ignore group messages and status/story updates
        if (from.endsWith('@g.us') || from === 'status@broadcast') return; 

        // --- 1. MULTI-FORMAT TEXT EXTRACTION ---
        // WhatsApp has many message types (Plain text, Buttons, Carousels). 
        // This block safely extracts the intended text or Button ID regardless of how the user replied.
        let text = msg.message?.conversation || msg.message?.extendedTextMessage?.text || '';
        
        if (msg.message?.interactiveResponseMessage) {
            // Extracts ID from Native Flow / Carousel buttons
            try {
                const params = JSON.parse(msg.message.interactiveResponseMessage.nativeFlowResponseMessage.paramsJson);
                if (params.id) text = params.id;
            } catch (e) {}
        } else if (msg.message?.templateButtonReplyMessage) {
            // Extracts ID from older Template Buttons
            text = msg.message.templateButtonReplyMessage.selectedId;
        } else if (msg.message?.buttonsResponseMessage) {
            // Extracts ID from older Standard Buttons
            text = msg.message.buttonsResponseMessage.selectedButtonId;
        }

        // 👉 SILENCE FILTER: Prevents the bot from processing empty system pings or unsupported media
        if (!text || text.trim().length === 0) return;

        const upperText = text.toUpperCase().trim();
        
        // Initialize user session if they are a first-time sender
        if (!userSessions[from]) userSessions[from] = { state: 'chat', cart: null };
        const session = userSessions[from];

        console.log(`[Input]: ${text} | State: ${session.state}`);

        // --- 2. UNIVERSAL CANCEL COMMAND ---
        // Allows the user to break out of a checkout loop at any time
        if (upperText === 'CANCEL' || upperText === 'CANCEL_CHECKOUT') {
            session.state = 'chat';
            session.cart = null;
            await sock.sendMessage(from, { text: "🛑 Action cancelled. How else can EllTek help you today?" });
            return;
        }

        // ==========================================
        // 💳 CHECKOUT STATE MACHINE
        // ==========================================
        // This isolates the user from the AI engine so they can safely enter checkout details

        // Step 1: User just clicked 'Buy' and sent their email
        if (session.state === 'awaiting_email') {
            session.email = text.trim();
            session.state = 'selecting_payment_method';
            await sendPaymentOptions(from, session.cart); // Trigger payment selection carousel
            return;
        }

        // Step 2a: User selected Paystack Web Link
        if (upperText === 'PAY_LINK') {
            await initiatePaystackLink(from, session);
            return;
        }

        // Step 2b: User selected Direct MoMo, ask for their number
        if (upperText === 'PAY_DIRECT') {
            session.state = 'awaiting_momo_number';
            await sock.sendMessage(from, { text: "📲 Please enter your MoMo number (e.g., 054XXXXXXX):" });
            return;
        }

        // Step 2c: User provided MoMo number, fire the Paystack Charge API
        if (session.state === 'awaiting_momo_number') {
            session.momoNumber = text.trim();
            session.state = 'chat'; // Reset state back to normal chat
            await sock.sendMessage(from, { text: "🔐 *SECURITY NOTE:* Your PIN is never saved or seen by EllTek or any 3rd party.\n\nPlease authorize the prompt on your phone and delete the OTP message immediately after." });
            await initiatePaystackCharge(from, session);
            return;
        }

        // Step 2d: User selected Manual Call
        if (upperText === 'PAY_MANUAL') {
            session.state = 'chat'; // Reset state
            await sock.sendMessage(from, { text: `📞 Please call our shop at *${SHOP_PHONE}* to complete your order for *${session.cart.name}*.` });
            return;
        }

        // ==========================================
        // 🛠️ HARDCODED UI COMMANDS (Fast path)
        // ==========================================
        // These process button clicks instantly without wasting API calls to the AI

        if (upperText === 'SHOW_CATALOG' || upperText === 'MENU') {
            const products = await fetchProducts();
            await sendProductCarousel(from, products.slice(0, 10), "Catalog 1/2: Featured Items");
            await sendProductCarousel(from, products.slice(10, 20), "Catalog 2/2: More Products");
            return; 
        }

        if (upperText.startsWith('VIEW_PRODUCT ') || (upperText.includes('VIEW') && /\d/.test(upperText))) {
            const productId = upperText.replace(/\D/g, ''); // Extract numbers only
            const product = (await fetchProducts()).find(p => p.id == productId);
            if (product) await sendProductDetails(from, product);
            return; 
        }

        if (upperText.startsWith('BUY ') || (upperText.includes('BUY') && /\d/.test(upperText))) {
            const productId = upperText.replace(/\D/g, '');
            const product = (await fetchProducts()).find(p => p.id == productId);
            if (product) {
                session.cart = product; // Save item to memory
                session.state = 'awaiting_email'; // Lock them into checkout state
                await sock.sendMessage(from, { text: `🛒 *${product.name}* (GH₵ ${product.price})\n\nPlease provide your email address for the payment receipt:` });
            }
            return; 
        }

        // ==========================================
        // 🧠 HYBRID AI INTERCEPTOR
        // ==========================================
        // If the message wasn't a button click or checkout step, pass it to the AI

        const products = await fetchProducts();
        
        // Show "typing..." indicator to make the bot feel human
        await sock.sendPresenceUpdate('composing', from); 
        
        // Get the response string from the AI Engine
        let aiResponse = await getAIResponse(text, products);
        
        // Variables to hold intercepted triggers
        let triggerCatalog = false;
        let viewId = null;
        let buyId = null;
        let filterIds = [];

        // --- TAG PARSING LOGIC ---
        // We look for specific tags the AI appended, set the flag, and remove the tag from the text 
        // so the user never sees the raw code.

        if (aiResponse.includes('[TRIGGER_CATALOG]')) {
            triggerCatalog = true;
            aiResponse = aiResponse.replace('[TRIGGER_CATALOG]', '').trim();
        }
        
        // Regex to catch comma-separated IDs: [TRIGGER_FILTER_1, 2, 3]
        const filterMatch = aiResponse.match(/\[TRIGGER_FILTER_([\d,\s]+)\]/);
        if (filterMatch) {
            filterIds = filterMatch[1].split(',').map(id => id.trim());
            aiResponse = aiResponse.replace(filterMatch[0], '').trim();
        }

        // Regex to catch specific view IDs: [TRIGGER_VIEW_5]
        const viewMatch = aiResponse.match(/\[TRIGGER_VIEW_(\d+)\]/);
        if (viewMatch) {
            viewId = viewMatch[1];
            aiResponse = aiResponse.replace(viewMatch[0], '').trim();
        }

        // Regex to catch specific buy IDs: [TRIGGER_BUY_2]
        const buyMatch = aiResponse.match(/\[TRIGGER_BUY_(\d+)\]/);
        if (buyMatch) {
            buyId = buyMatch[1];
            aiResponse = aiResponse.replace(buyMatch[0], '').trim();
        }

        // --- OUTPUT EXECUTION ---
        
        // Step 1: Send the clean, conversational text from the AI
        if (aiResponse.length > 0) {
            await sock.sendMessage(from, { text: aiResponse });
        }

        // Step 2: Fire the visual carousels based on the intercepted flags
        if (triggerCatalog) {
            await sendProductCarousel(from, products.slice(0, 10), "Catalog 1/2");
            await sendProductCarousel(from, products.slice(10, 20), "Catalog 2/2");
        } 
        else if (filterIds.length > 0) {
            // Build a custom carousel showing ONLY the items the AI selected
            const filteredProducts = products.filter(p => filterIds.includes(p.id.toString()));
            if (filteredProducts.length > 0) {
                await sendProductCarousel(from, filteredProducts.slice(0, 10), "Matching Products");
            }
        }
        else if (viewId) {
            const product = products.find(p => p.id == viewId);
            if (product) await sendProductDetails(from, product);
        }
        else if (buyId) {
            // Instantly transition the user into the checkout flow
            const product = products.find(p => p.id == buyId);
            if (product) {
                session.cart = product;
                session.state = 'awaiting_email'; 
                await sock.sendMessage(from, { text: `🛒 *${product.name}* (GH₵ ${product.price})\n\nPlease provide your email address for the payment receipt:` });
            }
        }
    });
}

// Builds and sends the multi-card carousel for selecting a payment method
async function sendPaymentOptions(to, product) {
    const media = await prepareWAMessageMedia({ image: { url: product.image_url } }, { upload: sock.waUploadToServer });

    const cards = [
        {
            header: { title: "🔗 Paystack Link", hasMediaAttachment: true, ...media },
            body: { text: `Securely pay GH₵ ${product.price} via your web browser.` },
            nativeFlowMessage: {
                buttons: [{ name: "quick_reply", buttonParamsJson: JSON.stringify({ display_text: "Select Web Link 🔗", id: "PAY_LINK" }) }]
            }
        },
        // ... (MoMo and Manual Call cards follow the same structure)
        {
            header: { title: "💳 Direct MoMo", hasMediaAttachment: true, ...media },
            body: { text: `Get a direct authorization prompt on your MoMo phone.` },
            nativeFlowMessage: {
                buttons: [{ name: "quick_reply", buttonParamsJson: JSON.stringify({ display_text: "Select Direct MoMo 📲", id: "PAY_DIRECT" }) }]
            }
        },
        {
            header: { title: "📞 Call to Order", hasMediaAttachment: true, ...media },
            body: { text: `Call EllTek directly to arrange your payment offline.` },
            nativeFlowMessage: {
                buttons: [{ name: "quick_reply", buttonParamsJson: JSON.stringify({ display_text: "Select Manual Call 📞", id: "PAY_MANUAL" }) }]
            }
        }
    ];

    const msg = generateWAMessageFromContent(to, { 
        viewOnceMessage: { 
            message: { 
                interactiveMessage: { 
                    body: { text: `*Checkout:* ${product.name}\nPlease swipe to select your payment method:` }, 
                    carouselMessage: { cards, messageVersion: 1 } 
                } 
            } 
        } 
    }, { userJid: sock.user.id });

    await sock.relayMessage(to, msg.message, { messageId: msg.key.id });
}

// Generates a standard web checkout link via Paystack
async function initiatePaystackLink(to, session) {
    try {
        const response = await axios.post('https://api.paystack.co/transaction/initialize', {
            email: session.email, 
            amount: session.cart.price * 100, // Paystack requires amounts in pesewas/kobo
            currency: "GHS", 
            callback_url: "https://elltek.com/callback" 
        }, { headers: { Authorization: `Bearer ${PAYSTACK_SECRET_KEY}` } });

        session.state = 'chat'; // Checkout complete, release the user
        await sock.sendMessage(to, { text: `🔗 *Payment Link Generated*\n\nOrder: ${session.cart.name}\nTotal: GH₵ ${session.cart.price}\n\nPay here: ${response.data.data.authorization_url}` });
    } catch (error) {
        await sock.sendMessage(to, { text: "❌ Invalid email. Please try again or type CANCEL." });
    }
}

// Uses Paystack's Charge API to push an MTN MoMo prompt directly to the user's phone
async function initiatePaystackCharge(to, session) {
    try {
        await axios.post('https://api.paystack.co/charge', {
            email: session.email, 
            amount: session.cart.price * 100, 
            currency: "GHS",
            mobile_money: { phone: session.momoNumber, provider: "mtn" }
        }, { headers: { Authorization: `Bearer ${PAYSTACK_SECRET_KEY}` } });

        await sock.sendMessage(to, { text: "⌛ Charge initiated. Please check your phone for the MoMo authorization prompt." });
    } catch (e) {
        await sock.sendMessage(to, { text: "❌ Direct charge failed. Please try the 'Paystack Link' option instead." });
    }
}

// Mock database function. For a live system, replace this with an Axios GET request to your Laravel backend.
async function fetchProducts() {
    const base = [
        { id: 1, name: "Nike Air Zoom", price: 450, description: "Premium running shoes. Ultra-lightweight with dynamic cushioning." },
        { id: 2, name: "Smart Watch Pro", price: 250, description: "Waterproof fitness tracker with a 7-day battery life and heart rate monitor." },
        { id: 3, name: "Wireless Earbuds", price: 150, description: "Active noise cancellation earbuds with deep bass and 24h play time." }
    ];
    // Generating dummy products to populate the 20-item catalog requirement
    const extra = Array.from({ length: 17 }, (_, i) => ({
        id: i + 4, name: `EllTek Item ${i + 4}`, price: 100 + (i * 20), description: "Official EllTek premium merchandise."
    }));
    return [...base, ...extra].map(p => ({ ...p, image_url: `https://placehold.co/600x400/103629/ffffff/png?text=${p.name.replace(/ /g, '+')}` }));
}

// Constructs and sends a horizontal scrolling carousel of product cards
async function sendProductCarousel(to, products, title) {
    // Process all images asynchronously
    const cards = await Promise.all(products.map(async (p) => {
        const media = await prepareWAMessageMedia({ image: { url: p.image_url } }, { upload: sock.waUploadToServer });
        return {
            body: { text: `*GH₵ ${p.price}*` },
            header: { title: p.name, hasMediaAttachment: true, ...media },
            nativeFlowMessage: {
                buttons: [
                    { name: "quick_reply", buttonParamsJson: JSON.stringify({ display_text: "View Details 👁️", id: `VIEW_PRODUCT ${p.id}` }) },
                    { name: "quick_reply", buttonParamsJson: JSON.stringify({ display_text: "Buy Now 🛒", id: `BUY ${p.id}` }) }
                ]
            }
        };
    }));
    // Generate the final WhatsApp protocol message object
    const msg = generateWAMessageFromContent(to, { viewOnceMessage: { message: { interactiveMessage: { body: { text: title }, carouselMessage: { cards, messageVersion: 1 } } } } }, { userJid: sock.user.id });
    await sock.relayMessage(to, msg.message, { messageId: msg.key.id });
}

// Constructs a single interactive card for detailed product viewing
async function sendProductDetails(to, product) {
    const media = await prepareWAMessageMedia({ image: { url: product.image_url } }, { upload: sock.waUploadToServer });
    const cards = [{
        header: { title: product.name, hasMediaAttachment: true, ...media },
        body: { text: `${product.description}\n\n*Price:* GH₵ ${product.price}` },
        nativeFlowMessage: {
            buttons: [
                { name: "quick_reply", buttonParamsJson: JSON.stringify({ display_text: "Buy Now 🛒", id: `BUY ${product.id}` }) },
                { name: "quick_reply", buttonParamsJson: JSON.stringify({ display_text: "Show Catalog 🔙", id: "SHOW_CATALOG" }) }
            ]
        }
    }];
    const msg = generateWAMessageFromContent(to, { viewOnceMessage: { message: { interactiveMessage: { body: { text: "Details:" }, carouselMessage: { cards, messageVersion: 1 } } } } }, { userJid: sock.user.id });
    await sock.relayMessage(to, msg.message, { messageId: msg.key.id });
}
